36 GPUconstexpr() BasicType Has3x3PeakMask = 1 << O2_GPU_PACKED_CHARGE_CHARGE_BITS;

52 class PackedCharge
53 {
54  public:
55  using BasicType = PackedChargeDefs::BasicType;
56 
57  PackedCharge() = default;
58  GPUdi() explicit PackedCharge(Charge q) : PackedCharge(q, false, false) {}
59  GPUdi() PackedCharge(Charge q, bool peak3x3, bool wasSplit)
60  {
61  using namespace PackedChargeDefs;
62 
63  mVal = q * Shift;
64  mVal = CAMath::Min(MaxVal, mVal); // ensure only lower 14 bits are set
65  mVal |= (peak3x3) ? Has3x3PeakMask : BasicType(0);
66  mVal |= (wasSplit) ? IsSplitMask : BasicType(0);
67  }
68 
69  GPUdi() Charge unpack() const { return Charge(mVal & PackedChargeDefs::ChargeMask) / PackedChargeDefs::Shift; }
70  GPUdi() bool has3x3Peak() const { return mVal & PackedChargeDefs::Has3x3PeakMask; }
71  GPUdi() bool isSplit() const { return mVal & PackedChargeDefs::IsSplitMask; }
72 
73  private:
74  BasicType mVal;
75 };
