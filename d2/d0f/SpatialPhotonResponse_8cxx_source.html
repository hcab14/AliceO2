35 void SpatialPhotonResponse::addPhoton(double x, double y, int nphotons)
36 {
37  const int xpixel = (int)(std::floor((x - mLowerX) * mInvLxOfCell));
38  const int ypixel = (int)(std::floor((y - mLowerY) * mInvLyOfCell));
39  if (nphotons < 0) {
40  std::cerr << "negative photon number\n";
41  return;
42  }
43  if (xpixel < 0 || xpixel >= mNx) {
44  std::cerr << "X-PIXEL OUT OF RANGE " << xpixel << " " << x << " , " << y << "\n";
45  return;
46  }
47  if (ypixel < 0 || ypixel >= mNy) {
48  std::cerr << "Y-PIXEL OUT OF RANGE " << ypixel << " " << x << " , " << y << "\n";
49  return;
50  }
51  mImageData[xpixel][ypixel] += nphotons;
52  mPhotonSum += nphotons;
53 }
