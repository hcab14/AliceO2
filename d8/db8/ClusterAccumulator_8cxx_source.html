29  bool wasSplitInPad = mSplitInPad >= MIN_SPLIT_NUM;

36  cn.qmax = d.charge;

38  deprecated::cnSetTimeFlags(&cn, mTimeMean, flags);

44 GPUd() void ClusterAccumulator::update(Charge splitCharge, Delta dp, Delta dt)
45 {
46  mQtot += splitCharge;
47  mPadMean += splitCharge * dp;
48  mTimeMean += splitCharge * dt;
49  mPadSigma += splitCharge * dp * dp;
50  mTimeSigma += splitCharge * dt * dt;
51 }

53 GPUd() Charge ClusterAccumulator::updateInner(PackedCharge charge, Delta dp, Delta dt)
54 {
55  Charge q = charge.unpack();
56 
57  update(q, dp, dt);
58 
59  bool split = charge.isSplit();

62

66 GPUd() Charge ClusterAccumulator::updateOuter(PackedCharge charge, Delta dp, Delta dt)

77

81 GPUd() void ClusterAccumulator::finalize(const deprecated::Digit& myDigit)
82 {
83  mQtot += myDigit.charge;
84  if (mQtot == 0) {
85  return; // TODO: Why does this happen?
86  }
87 
88  mPadMean /= mQtot;
89  mTimeMean /= mQtot;
90  mPadSigma /= mQtot;
91  mTimeSigma /= mQtot;
92 
93  mPadSigma = sqrt(mPadSigma - mPadMean * mPadMean);
94  mTimeSigma = sqrt(mTimeSigma - mTimeMean * mTimeMean);
95 
96  mPadMean += myDigit.pad;
97  mTimeMean += myDigit.time;
98 
99 #if defined(CORRECT_EDGE_CLUSTERS)
100  if (CfUtils::isAtEdge(myDigit)) {
101  float s = (myDigit->pad < 2) ? 1.f : -1.f;
102  bool c = s * (mPadMean - myDigit->pad) > 0.f;
103  mPadMean = (c) ? myDigit->pad : mPadMean;
104  }
105 #endif
106 }
