117 GPUd() char Deconvolution::countPeaksAroundDigit(
118  const GlobalPad gpad,
119  const Timestamp time,
120  GPUglobalref() const uchar* peakMap)
121 {
122  char peakCount = 0;
123 
124  uchar aboveThreshold = 0;
125  for (uchar i = 0; i < 8; i++) {
126  Delta2 d = CfConsts::InnerNeighbors[i];
127  Delta dp = d.x;
128  Delta dt = d.y;
129 
130  uchar p = IS_PEAK(peakMap, gpad + dp, time + dt);
131  peakCount += GET_IS_PEAK(p);
132  aboveThreshold |= GET_IS_ABOVE_THRESHOLD(p) << i;
133  }
134 
135  if (peakCount > 0) {
136  return peakCount;
137  }
138 
139  for (uchar i = 0; i < 16; i++) {
140  Delta2 d = CfConsts::OuterNeighbors[i];
141  Delta dp = d.x;
142  Delta dt = d.y;
143 
144  if (CfUtils::innerAboveThresholdInv(aboveThreshold, i)) {
145  peakCount -= GET_IS_PEAK(IS_PEAK(peakMap, gpad + dp, time + dt));
146  }
147  }
148 
149  return peakCount;
150 }
