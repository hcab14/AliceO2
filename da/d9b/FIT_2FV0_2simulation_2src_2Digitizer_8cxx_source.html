40 void Digitizer::init()
41 {
42  LOG(INFO) << "V0Digitizer::init -> start = ";
43 
44  // Zero iterators for random rings
45  mIteratorScintDelay = 0;
46  mIteratorGainVar = 0;
47  mIteratorSignalShape = 0;
48 
49  const float y = DigitizationParameters::mPmtTransitTime;
50  format pmresponse("(x+%1%)*(x+%2%)*TMath::Exp(-(x+%3%)*(x+%4%)/(%5% * %6%))");
51  pmresponse % y % y % y % y % y % y;
52  std::string pmtResponseFormula = pmresponse.str();
53 
54  mNBins = 2000; //Will be computed using detector set-up from CDB
55  mBinSize = 25.0 / 256.0; //Will be set-up from CDB
56 
57  for (Int_t i = 0; i < DigitizationParameters::NCHANNELS; i++) {
58  mPmtChargeVsTime[i].resize(mNBins);
59  }
60 
61  // Initialize function describing the PMT time response
62  std::unique_ptr<o2::base::CachingTF1> mPmtResponse = std::make_unique<o2::base::CachingTF1>("mPmtResponse", pmtResponseFormula.c_str(), -y, 2 * y);
63  mPmtResponse->SetNpx(100);
64  mPmtTimeIntegral = mPmtResponse->Integral(-DigitizationParameters::mPmtTransitTime,
65  2. * DigitizationParameters::mPmtTransitTime);
66 
67  // Prefill random ring with time delay from scintillator
68  for (int i = 0; i < DigitizationParameters::HIT_RANDOM_RING_SIZE; i++) {
69  mScintillatorDelay.at(i) = gRandom->Gaus(0, DigitizationParameters::mIntrinsicTimeRes);
70  }
71 
72  // Initialize function describing PMT response to the single photoelectron
73  std::unique_ptr<o2::base::CachingTF1> mSinglePhESpectrum = std::make_unique<o2::base::CachingTF1>("mSinglePhESpectrum", this, &Digitizer::SinglePhESpectrum, DigitizationParameters::photoelMin, DigitizationParameters::photoelMax, 0);
74  Float_t meansPhE = mSinglePhESpectrum->Mean(DigitizationParameters::photoelMin, DigitizationParameters::photoelMax);
75  // Prefill random ring with variation of gain with respect to its mean value
76  for (int i = 0; i < DigitizationParameters::PHE_RANDOM_RING_SIZE; i++) {
77  mGainVar.at(i) = mSinglePhESpectrum->GetRandom(DigitizationParameters::photoelMin, DigitizationParameters::photoelMax) / meansPhE;
78  }
79 
80  std::unique_ptr<o2::base::CachingTF1> signalShape;
81  signalShape = std::make_unique<o2::base::CachingTF1>("signalShape", "crystalball", 0, 300);
82  signalShape->SetParameters(1,
83  DigitizationParameters::mShapeSigma,
84  DigitizationParameters::mShapeSigma,
85  DigitizationParameters::mShapeAlpha,
86  DigitizationParameters::mShapeN);
87  // Prefill random ring with time delay of photoelectron hit coming from measured signal shape
88  for (int i = 0; i < DigitizationParameters::PHE_RANDOM_RING_SIZE; i++) {
89  mSignalShape.at(i) = signalShape->GetRandom(0, mBinSize * Float_t(mNBins));
90  }
91 
92  LOG(INFO) << "V0Digitizer::init -> finished";
93 }

95 void Digitizer::process(const std::vector<o2::fv0::Hit>& hits,
96  std::vector<o2::fv0::BCData>& digitsBC,
97  std::vector<o2::fv0::ChannelData>& digitsCh,
98  o2::dataformats::MCTruthContainer<o2::fv0::MCLabel>& labels)
99 {
100  LOG(INFO) << "[FV0] Digitizer::process(): begin with " << hits.size() << " hits";
101 
102  auto sorted_hits{hits};
103  std::sort(sorted_hits.begin(), sorted_hits.end(), [](o2::fv0::Hit const& a, o2::fv0::Hit const& b) {
104  return a.GetTrackID() < b.GetTrackID();
105  });
106 
107  Int_t parentIdPrev = -10;
108  for (auto& hit : sorted_hits) {
109  int detId = hit.GetDetectorID();
110  const double hitEdep = hit.GetHitValue() * 1e3; //convert to MeV
111 
112  // TODO: check how big is inaccuracy if more than 1 'below-threshold' particles hit the same detector cell
113  if (hitEdep < DigitizationParameters::singleMipThreshold) {
114  continue;
115  }
116 
117  Double_t const nPhotons = hitEdep * DigitizationParameters::N_PHOTONS_PER_MEV;
118  Int_t const nPhE = SimulateLightYield(detId, nPhotons);
119  Float_t const t = mScintillatorDelay.at(mIteratorScintDelay) + hit.GetTime() * 1e9;
120  Float_t charge = TMath::Qe() * DigitizationParameters::mPmtGain * mBinSize / mPmtTimeIntegral;
121  for (Int_t iPhE = 0; iPhE < nPhE; ++iPhE) {
122  Float_t const tPhE = t + mSignalShape.at(mIteratorSignalShape);
123  Int_t const firstBin = TMath::Max((Int_t)0, (Int_t)((tPhE - DigitizationParameters::mPmtTransitTime) / mBinSize));
124  Int_t const lastBin = TMath::Min((Int_t)mNBins - 1, (Int_t)((tPhE + 2. * DigitizationParameters::mPmtTransitTime) / mBinSize));
125  for (Int_t iBin = firstBin; iBin <= lastBin; ++iBin) {
126  Float_t const tempT = mBinSize * (0.5 + iBin) - tPhE;
127  mPmtChargeVsTime.at(detId).at(iBin) += mGainVar.at(mIteratorGainVar) * charge * PmtResponse(tempT);
128  } // time-bin loop
129 
130  // RandomRing servicing (perhaps a separate class could do better)
131  mIteratorGainVar++;
132  if (mIteratorGainVar >= mGainVar.size()) {
133  mIteratorGainVar = 0;
134  }
135  mIteratorSignalShape++;
136  if (mIteratorSignalShape >= mSignalShape.size()) {
137  mIteratorSignalShape = 0;
138  }
139  } //photo electron loop
140  mIteratorScintDelay++;
141  if (mIteratorScintDelay >= mScintillatorDelay.size()) {
142  mIteratorScintDelay = 0;
143  }
144 
145  // Charged particles in MCLabel
146  Int_t parentId = hit.GetTrackID();
147  if (parentId != parentIdPrev) {
148  mMCLabels.emplace_back(parentId, mEventId, mSrcId, detId);
149  parentIdPrev = parentId;
150  }
151  } //hit loop
152 
153  // Sum charge of all time bins to get total charge collected for a given channel
154  int first = digitsCh.size(), nStored = 0;
155  for (Int_t ipmt = 0; ipmt < DigitizationParameters::NCHANNELS; ++ipmt) {
156  Float_t totalCharge = 0;
157  for (Int_t iTimeBin = 0; iTimeBin < mNBins; ++iTimeBin) {
158  Float_t timeBinCharge = mPmtChargeVsTime[ipmt][iTimeBin] / DigitizationParameters::CHARGE_PER_ADC;
159  totalCharge += timeBinCharge;
160  }
161  digitsCh.emplace_back(ipmt, SimulateTimeCfd(ipmt), totalCharge);
162  nStored++;
163  }
164 
165  // Send MClabels and digitsBC to storage
166  int nBC = digitsBC.size();
167  digitsBC.emplace_back(first, nStored, mIntRecord);
168  for (const auto& lbl : mMCLabels) {
169  labels.addElement(nBC, lbl);
170  }
171 }

177 Int_t Digitizer::SimulateLightYield(Int_t pmt, Int_t nPhot)
178 {
179  const Float_t epsilon = 0.0001;
180  const Float_t p = DigitizationParameters::mLightYield * DigitizationParameters::mPhotoCathodeEfficiency;
181  if ((fabs(1.0f - p) < epsilon) || nPhot == 0) {
182  return nPhot;
183  }
184  const Int_t n = Int_t(nPhot < 100
185  ? gRandom->Binomial(nPhot, p)
186  : gRandom->Gaus(p * nPhot + 0.5, TMath::Sqrt(p * (1 - p) * nPhot)));
187  return n;
188 }

191 Float_t Digitizer::SimulateTimeCfd(Int_t channel)
192 {
193  //std::fill(mPmtChargeVsTimeCfd.begin(), mPmtChargeVsTimeCfd.end(), 0);
194  Float_t timeCfd = -1024;
195  Int_t const binShift = TMath::Nint(DigitizationParameters::mTimeShiftCfd / mBinSize);
196  Float_t sigPrev = -mPmtChargeVsTime[channel][0];
197  for (Int_t iTimeBin = 1; iTimeBin < mNBins; ++iTimeBin) {
198  //if (mPmtChargeVsTime[channel][iTimeBin] != 0) std::cout << mPmtChargeVsTime[channel][iTimeBin] / parameters.CHARGE_PER_ADC << ", ";
199  Float_t sigCurrent = (iTimeBin >= binShift
200  ? 5.0 * mPmtChargeVsTime[channel][iTimeBin - binShift] - mPmtChargeVsTime[channel][iTimeBin]
201  : -mPmtChargeVsTime[channel][iTimeBin]);
202  if (sigPrev < 0 && sigCurrent >= 0) {
203  timeCfd = mBinSize * Float_t(iTimeBin);
204  //std::cout<<timeCfd<<std::endl;
205  break;
206  }
207  sigPrev = sigCurrent;
208  }
209  return timeCfd;
210 }

213 Float_t Digitizer::PmtResponse(Float_t x)
214 {
215  // this function describes the PMT time response to a single photoelectron
216  x += DigitizationParameters::mPmtTransitTime;
217  Float_t x2 = x * x;
218  return x2 * TMath::Exp(-x2 / DigitizationParameters::mPmtTransitTime2);
219 }

222 Double_t Digitizer::SinglePhESpectrum(Double_t* x, Double_t*)
223 {
224  // x -- number of photo-electrons emitted from the first dynode
225  // this function describes the PMT amplitude response to a single photoelectron
226  if (x[0] < 0)
227  return 0;
228  return (TMath::Poisson(x[0], DigitizationParameters::mPmtNbOfSecElec) +
229  DigitizationParameters::mPmtTransparency * TMath::Poisson(x[0], 1.0));
230 }
