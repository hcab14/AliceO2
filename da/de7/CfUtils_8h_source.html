47  static GPUdi() ushort partition(GPUTPCClusterFinderKernels::GPUTPCSharedMemory& smem, ushort ll, bool pred, ushort partSize, ushort* newPartSize)
48  {
49  bool participates = ll < partSize;
50 
51  ushort lpos = work_group_scan_inclusive_add((int)(!pred && participates));
52 
53  ushort part = work_group_broadcast(lpos, SCRATCH_PAD_WORK_GROUP_SIZE - 1);
54 
55  lpos -= 1;
56  ushort pos = (participates && !pred) ? lpos : part;
57 
58  *newPartSize = part;
