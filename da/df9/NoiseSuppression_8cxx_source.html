99 GPUd() void NoiseSuppression::checkForMinima(
100  float q,
101  float epsilon,
102  PackedCharge other,
103  int pos,
104  ulong* minimas,
105  ulong* bigger)
106 {
107  float r = other.unpack();
108 
109  ulong isMinima = (q - r > epsilon);
110  *minimas |= (isMinima << pos);
111 
112  ulong lq = (r > q);
113  *bigger |= (lq << pos);
114 }

147 GPUd() void NoiseSuppression::findMinima(
148  GPUglobalref() const PackedCharge* chargeMap,
149  const GlobalPad gpad,
150  const Timestamp time,
151  const float q,
152  const float epsilon,
153  ulong* minimas,
154  ulong* bigger)
155 {
156  *minimas = 0;
157  *bigger = 0;
158 
159  for (int i = 0; i < NOISE_SUPPRESSION_NEIGHBOR_NUM; i++) {
160  Delta2 d = CfConsts::NoiseSuppressionNeighbors[i];
161  Delta dp = d.x;
162  Delta dt = d.y;
163 
164  PackedCharge other = CHARGE(chargeMap, gpad + dp, time + dt);
165 
166  checkForMinima(q, epsilon, other, i, minimas, bigger);
167  }
168 }

170 GPUd() ulong NoiseSuppression::findPeaks(
171  GPUglobalref() const uchar* peakMap,
172  const GlobalPad gpad,
173  const Timestamp time,
174  bool debug)
175 {
176  ulong peaks = 0;
177  if (debug) {
178  printf("Looking around %d, %d\n", gpad, time);
179  }
180  for (int i = 0; i < NOISE_SUPPRESSION_NEIGHBOR_NUM; i++) {
181  Delta2 d = CfConsts::NoiseSuppressionNeighbors[i];
182  Delta dp = d.x;
183  Delta dt = d.y;
184 
185  uchar p = IS_PEAK(peakMap, gpad + dp, time + dt);
186 
187  if (debug) {
188  printf("%d, %d: %d\n", dp, dt, p);
189  }
190 
191  peaks |= (ulong(GET_IS_PEAK(p)) << i);
192  }
193 
194  return peaks;
195 }

197 GPUd() bool NoiseSuppression::keepPeak(
198  ulong minima,
199  ulong peaks)
200 {
201  bool keepMe = true;
202 
203  for (int i = 0; i < NOISE_SUPPRESSION_NEIGHBOR_NUM; i++) {
204  bool otherPeak = (peaks & (ulong(1) << i));
205  bool minimaBetween = (minima & CfConsts::NoiseSuppressionMinima[i]);
206 
207  keepMe &= (!otherPeak || minimaBetween);
208  }
209 
210  return keepMe;
211 }
