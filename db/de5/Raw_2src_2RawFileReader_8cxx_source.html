63 size_t RawFileReader::LinkData::getLargestSuperPage() const
64 {
65  // estimate largest super page size
66  size_t szMax = 0, szLast = 0;
67  for (const auto& bl : blocks) {
68  if (bl.startSP) { // account previous SPage and start accumulation of the next one
69  if (szLast > szMax) {
70  szMax = szLast;
71  }
72  szLast = 0;
73  }
74  szLast += bl.size;
75  }
76  return szLast > szMax ? szLast : szMax;
77 }

97 bool RawFileReader::LinkData::preprocessCRUPage(const RDH& rdh, bool newSPage)
98 {
99  // account RDH in statistics
100  bool ok = true;
101  bool newTF = false, newHB = false;
102 
103  if (rdh.feeId != rdhl.feeId) { // make sure links with different FEEID were not assigned same subspec
104  LOG(ERROR) << "Same SubSpec is found for Links with different RDH.feeId";
105  printf("old RDH assigned SubSpec=0x%-8d:\n", subspec);
106  o2::raw::HBFUtils::dumpRDH(rdhl);
107  printf("new RDH assigned SubSpec=0x%-8d:\n", subspec);
108  o2::raw::HBFUtils::dumpRDH(rdh);
109  LOG(FATAL) << "Critical error, aborting";
110  ok = false;
111  nErrors++;
112  }
113 
114  if (rdh.pageCnt == 0) {
115  newTF = (rdh.triggerType & o2::trigger::TF);
116  newHB = (rdh.triggerType & (o2::trigger::ORBIT | o2::trigger::HB)) == (o2::trigger::ORBIT | o2::trigger::HB);
117  } else if (reader->mCheckErrors) {
118  // check increasing pageCnt
119  if (nCRUPages && (rdh.pageCnt != ((rdhl.pageCnt + 1) & 0xffff))) { // skip for very 1st page
120  LOG(ERROR) << "RDH.pageCnt=" << int(rdh.pageCnt) << " is not +1 wrt previous RDH.pageCnt=" << int(rdhl.pageCnt);
121  ok = false;
122  nErrors++;
123  }
124  }
125 
126  if (reader->mCheckErrors) {
127  if (nCRUPages) {
128  // check increasing (or wrapping) packetCounter
129  if ((rdh.packetCounter != ((rdhl.packetCounter + 1) & 0xff))) { // skip for very 1st page
130  LOG(ERROR) << "RDH.packetCounter=" << int(rdh.packetCounter)
131  << " is not + 1 wrt previous RDH.packetCounter=" << int(rdhl.packetCounter);
132  ok = false;
133  nErrors++;
134  }
135  // check if number of HBFs in the TF is as expected
136  if (newTF) {
137  if (nHBFinTF != reader->mNominalHBFperTF) {
138  LOG(ERROR) << "Number of HBFs in TF " << nHBFinTF << " differs from expected " << reader->mNominalHBFperTF;
139  ok = false;
140  nErrors++;
141  }
142  nHBFinTF = 0; // reset
143  }
144 
145  } else { // make sure data starts with TF and HBF
146  if (!newTF || !newHB || rdh.pageCnt != 0) {
147  LOG(ERROR) << "Very 1st RDH of the data does not start with TF or new HBF";
148  ok = false;
149  nErrors++;
150  }
151  }
152  }
153 
154  if (newHB) {
155  if (reader->mCheckErrors) {
156  nHBFinTF++;
157  if (rdh.stop) {
158  LOG(ERROR) << "Stop of HBF #" << nHBFrames << " is found at 1st page";
159  ok = false;
160  nErrors++;
161  }
162  if (openHB) {
163  LOG(ERROR) << "New HBF#" << nHBFrames << " starts while previous HBF was not closed";
164  ok = false;
165  nErrors++;
166  }
167  if (nCRUPages && !checkIRIncrement(rdh, rdhl)) { // skip check for the very 1st RDH
168  ok = false;
169  nErrors++;
170  }
171  } // end of check errors
172  openHB = true;
173  nHBFrames++;
174  }
175  if (rdh.stop) {
176  openHB = false;
177  }
178 
179  if (newTF || newSPage) {
180  auto& bl = blocks.emplace_back(reader->mCurrentFileID, reader->mPosInFile);
181  if (newTF) {
182  nTimeFrames++;
183  bl.startTF = true;
184  if (reader->mCheckErrors) {
185  if (reader->mMultiLinkFile && !newSPage) {
186  LOG(ERROR) << "New TF#" << nTimeFrames << " does not start new from new superpage";
187  ok = false;
188  nErrors++;
189  }
190  /* // this test seems to be meaningless
191  if (!newHB) {
192  LOG(ERROR) << "New TF#" << nTimeFrames << " does not start with new HBF";
193  ok = false;
194  nErrors++;
195  }
196  */
197  } // end of check errors
198  }
199  if (newSPage) {
200  nSPages++;
201  bl.startSP = true;
202  bl.startHB = true;
203  }
204  }
205  rdhl = rdh;
206  nCRUPages++;
207  if (!ok) {
208  LOG(ERROR) << " ^^^Problem(s) was encountered at offset " << reader->mPosInFile << " of file " << reader->mCurrentFileID;
209  if (reader->mVerbosity > 0) {
210  o2::raw::HBFUtils::printRDH(rdh);
211  }
212  } else {
213  if (reader->mVerbosity > 1) {
214  o2::raw::HBFUtils::printRDH(rdh);
215  }
216  }
217  return true;
218 }

225 bool RawFileReader::checkRDH(const o2::header::RAWDataHeaderV4& rdh) const
226 {
227  // check if rdh conforms with RDH4 fields
228  bool ok = true;
229  if (rdh.version != 4) {
230  LOG(ERROR) << "RDH version 4 is expected instead of " << int(rdh.version);
231  ok = false;
232  }
233  if (rdh.headerSize != 64) {
234  LOG(ERROR) << "RDH with header size of 64 B is expected instead of " << int(rdh.headerSize);
235  ok = false;
236  }
237  if (rdh.memorySize < 64 || rdh.offsetToNext < 64) {
238  LOG(ERROR) << "RDH expected to have memory size/offset to next >= 64 B instead of "
239  << int(rdh.memorySize) << '/' << int(rdh.offsetToNext);
240  ok = false;
241  }
242  if (rdh.zero0 || rdh.word3 || rdh.zero41 || rdh.zero42 || rdh.word5 || rdh.zero6 || rdh.word7) {
243  LOG(ERROR) << "Some reserved fields of RDH v4 are not empty";
244  ok = false;
245  }
246  if (!ok) {
247  o2::raw::HBFUtils::dumpRDH(rdh);
248  }
249  return ok;
250 }

281 int RawFileReader::getLinkLocalID(const RDH& rdh)
282 {
283  // get id of the link subspec. in the parser (create entry if new)
284  LinkSpec_t subspec = getSubSpec(rdh.cruID, rdh.linkID, rdh.endPointID);
285  auto entryMap = mLinkEntries.find(subspec);
286  if (entryMap == mLinkEntries.end()) { // need to register a new link
287  int n = mLinkEntries.size();
288  mLinkEntries[subspec] = n;
289  auto& lnk = mLinksData.emplace_back(rdh, this);
290  lnk.subspec = subspec;
291  return n;
292  }
293  return entryMap->second;
294 }

296 //_____________________________________________________________________
297 bool RawFileReader::preprocessFile(int ifl)
298 {
299  // preprocess file, check RDH data, build statistics
300  FILE* fl = mFiles[ifl];
301  mCurrentFileID = ifl;
302  RDH rdh;
303 
304  LinkSpec_t subspecPrev = 0xffffffff;
305  int lIDPrev = -1;
306  mMultiLinkFile = false;
307  rewind(fl);
308  long int nr = 0;
309  mPosInFile = 0;
310  int nRDHread = 0;
311  bool ok = true;
312  int readBytes = sizeof(RDH);
313  while ((nr = fread(&rdh, 1, readBytes, fl))) {
314  if (nr < readBytes) {
315  LOG(ERROR) << "EOF was unexpected, only " << nr << " bytes were read for RDH";
316  ok = false;
317  break;
318  }
319  if (!(ok = checkRDH(rdh))) {
320  break;
321  }
322  nRDHread++;
323  LinkSpec_t subspec = getSubSpec(rdh.cruID, rdh.linkID, rdh.endPointID);
324  int lID = lIDPrev;
325  if (subspec != subspecPrev) { // link has changed
326  subspecPrev = subspec;
327  if (lIDPrev != -1) {
328  mMultiLinkFile = true;
329  }
330  lID = getLinkLocalID(rdh);
331  }
332  bool newSPage = lID != lIDPrev;
333  mLinksData[lID].preprocessCRUPage(rdh, newSPage);
334  if (newSPage && lIDPrev != -1) { // flag the end of the block of previosly processed link
335  auto& prevBlock = mLinksData[lIDPrev].blocks.back();
336  prevBlock.size = mPosInFile - prevBlock.offset;
337  }
338  //
339  mPosInFile += rdh.offsetToNext;
340  if (fseek(fl, mPosInFile, SEEK_SET)) {
341  break;
342  }
343  lIDPrev = lID;
344  }
345  mPosInFile = ftell(fl);
346  if (lIDPrev != -1) { // close last block
347  auto& lastBlock = mLinksData[lIDPrev].blocks.back();
348  lastBlock.size = mPosInFile - lastBlock.offset;
349  }
350 
351  LOG(INFO) << "#" << mCurrentFileID << ": " << mPosInFile << " bytes scanned, " << nRDHread << " RDH read for "
352  << mLinkEntries.size() << " links from " << mFileNames[mCurrentFileID];
353  return ok;
354 }

401 //_____________________________________________________________________
402 bool RawFileReader::init()
403 {
404  int nf = mFiles.size();
405  bool ok = true;
406  for (int i = 0; i < nf; i++) {
407  ok &= preprocessFile(i);
408  }
409  mOrderedIDs.resize(mLinksData.size());
410  for (int i = mLinksData.size(); i--;) {
411  mOrderedIDs[i] = i;
412  }
413  std::sort(mOrderedIDs.begin(), mOrderedIDs.end(),
414  [& links = mLinksData](int a, int b) { return links[a].subspec < links[b].subspec; });
415 
416  size_t maxSP = 0, maxTF = 0;
417  printf("\nSummary of preprocessing:\n");
418  for (int i = 0; i < int(mLinksData.size()); i++) {
419  const auto& link = getLink(i);
420  auto msp = link.getLargestSuperPage();
421  auto mtf = link.getLargestTF();
422  if (maxSP < msp) {
423  maxSP = msp;
424  }
425  if (maxTF < mtf) {
426  maxTF = mtf;
427  }
428  printf("L%-4d| ", i);
429  link.print(mVerbosity);
430  if (msp > mNominalSPageSize) {
431  printf(" Attention: largest superpage %zu B exceeds expected %d B\n",
432  msp, mNominalSPageSize);
433  }
434  }
435  printf("Largest super-page: %zu B, largest TF: %zu B\n", maxSP, maxTF);
436  if (!mCheckErrors) {
437  printf("Detailed data format check was disabled\n");
438  }
439  mInitDone = true;
440 
441  return ok;
442 }
