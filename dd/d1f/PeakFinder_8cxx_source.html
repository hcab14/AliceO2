24 GPUd() bool PeakFinder::isPeakScratchPad(
25  GPUTPCClusterFinderKernels::GPUTPCSharedMemory& smem,
26  const Digit* digit,
27  ushort N,
28  GPUglobalref() const PackedCharge* chargeMap,
29  GPUsharedref() ChargePos* posBcast,
30  GPUsharedref() PackedCharge* buf)
31 {
32  ushort ll = get_local_id(0);
33 
34  const Timestamp time = digit->time;
35  const Row row = digit->row;
36  const Pad pad = digit->pad;
37 
38  const GlobalPad gpad = Array2D::tpcGlobalPadIdx(row, pad);
39  ChargePos pos = {gpad, time};
40 
41  bool belowThreshold = (digit->charge <= QMAX_CUTOFF);
42 
43  ushort lookForPeaks;
44  ushort partId = CfUtils::partition(
45  smem,
46  ll,
47  belowThreshold,
48  SCRATCH_PAD_WORK_GROUP_SIZE,
49  &lookForPeaks);
50 
51  if (partId < lookForPeaks) {
52  posBcast[partId] = pos;
53  }
54  GPUbarrier();
55 
56  CfUtils::fillScratchPad_PackedCharge(
57  chargeMap,
58  lookForPeaks,
59  SCRATCH_PAD_WORK_GROUP_SIZE,
60  ll,
61  0,
62  N,
63  CfConsts::InnerNeighbors,
64  posBcast,
65  buf);
66 
67  if (belowThreshold) {
68  return false;
69  }
70 
71  bool peak = true;
72  for (ushort i = 0; i < N; i++) {
73  Charge q = buf[N * partId + i].unpack();
74  peak &= (digit->charge > q) || (CfConsts::InnerTestEq[i] && digit->charge == q);
75  }
76 
77  return peak;
78 }

80 GPUd() bool PeakFinder::isPeak(
81  const Digit* digit,
82  GPUglobalref() const PackedCharge* chargeMap)
83 {
84  if (digit->charge <= QMAX_CUTOFF) {
85  return false;
86  }
87 
88  const Charge myCharge = digit->charge;
89  const Timestamp time = digit->time;
90  const Row row = digit->row;
91  const Pad pad = digit->pad;
92 
93  const GlobalPad gpad = Array2D::tpcGlobalPadIdx(row, pad);
94 
95  bool peak = true;
96 
97 #define CMP_NEIGHBOR(dp, dt, cmpOp) \
98  do { \
99  PackedCharge p = CHARGE(chargeMap, gpad + dp, time + dt); \
100  const Charge otherCharge = p.unpack(); \
101  peak &= (otherCharge cmpOp myCharge); \
102  } while (false)
103 
104 #define CMP_LT CMP_NEIGHBOR(-1, -1, <=)
105 #define CMP_T CMP_NEIGHBOR(-1, 0, <=)
106 #define CMP_RT CMP_NEIGHBOR(-1, 1, <=)
107 
108 #define CMP_L CMP_NEIGHBOR(0, -1, <=)
109 #define CMP_R CMP_NEIGHBOR(0, 1, <)
110 
111 #define CMP_LB CMP_NEIGHBOR(1, -1, <)
112 #define CMP_B CMP_NEIGHBOR(1, 0, <)
113 #define CMP_RB CMP_NEIGHBOR(1, 1, <)
114 
115 #if defined(CHARGEMAP_TILING_LAYOUT)
116  CMP_LT;
117  CMP_T;
118  CMP_RT;
119  CMP_R;
120  CMP_RB;
121  CMP_B;
122  CMP_LB;
123  CMP_L;
124 #else
125  CMP_LT;
126  CMP_T;
127  CMP_RT;
128  CMP_L;
129  CMP_R;
130  CMP_LB;
131  CMP_B;
132  CMP_RB;
133 #endif
134 
135 #undef CMP_LT
136 #undef CMP_T
137 #undef CMP_RT
138 #undef CMP_L
139 #undef CMP_R
140 #undef CMP_LB
141 #undef CMP_B
142 #undef CMP_RB
143 #undef CMP_NEIGHBOR
144 
145  return peak;
146 }

148 GPUd() void PeakFinder::findPeaksImpl(int nBlocks, int nThreads, int iBlock, int iThread, GPUTPCClusterFinderKernels::GPUTPCSharedMemory& smem,
149  GPUglobalref() const PackedCharge* chargeMap,
150  GPUglobalref() const Digit* digits,
151  uint digitnum,
152  GPUglobalref() uchar* isPeakPredicate,
153  GPUglobalref() uchar* peakMap)
154 {
155  size_t idx = get_global_id(0);
156 
157  // For certain configurations dummy work items are added, so the total
158  // number of work items is dividable by 64.
159  // These dummy items also compute the last digit but discard the result.
160  Digit myDigit = digits[CAMath::Min(idx, (size_t)(digitnum - 1))];
161 
162  uchar peak;
163 #if defined(BUILD_CLUSTER_SCRATCH_PAD)
164  peak = isPeakScratchPad(smem, &myDigit, SCRATCH_PAD_SEARCH_N, chargeMap, smem.search.posBcast, smem.search.buf);
165 #else
166  peak = isPeak(&myDigit, chargeMap);
167 #endif
168 
169  // Exit early if dummy. See comment above.
170  bool iamDummy = (idx >= digitnum);
171  if (iamDummy) {
172  return;
173  }
174 
175  isPeakPredicate[idx] = peak;
176 
177  const GlobalPad gpad = Array2D::tpcGlobalPadIdx(myDigit.row, myDigit.pad);
178 
179  IS_PEAK(peakMap, gpad, myDigit.time) =
180  ((myDigit.charge > CHARGE_THRESHOLD) << 1) | peak;
181 }
