131 inline RawFileReader::LinkSpec_t RawFileReader::getSubSpec(uint16_t cru, uint8_t link, uint8_t endpoint) const
132 {
133  // define subspecification as in DataDistribution
134  int linkValue = (RawFileReader::LinkSpec_t(link) + 1) << (endpoint == 1 ? 8 : 0);
135  return (RawFileReader::LinkSpec_t(cru) << 16) | linkValue;
136 }

139 inline bool RawFileReader::LinkData::checkIRIncrement(const o2::header::RAWDataHeaderV4& rdhNew,
140  const o2::header::RAWDataHeaderV4& rdhOld) const
141 {
142  // check orbit/bc increment
143  if ((rdhNew.heartbeatBC != rdhOld.heartbeatBC) || (rdhNew.heartbeatOrbit != rdhOld.heartbeatOrbit + 1)) {
144  LOG(ERROR) << "New HB orbit/bc=" << int(rdhNew.heartbeatOrbit) << '/' << int(rdhNew.heartbeatBC)
145  << " is not incremented by 1 orbit wrt Old HB orbit/bc="
146  << int(rdhOld.heartbeatOrbit) << '/' << int(rdhOld.heartbeatBC);
147  return false;
148  }
149  return true;
150 }
